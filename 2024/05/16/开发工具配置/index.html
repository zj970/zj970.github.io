通用配置

JVM 配置


# 初始堆内存大小：设置为16GB
-Xms16g

# 最大堆内存大小：调整为26GB（解决压缩指针限制）
-Xmx26g

# 元空间最大大小：1GB
-XX:MaxMetaspaceSize=1024m

# JIT编译代码缓存区大小：1GB
-XX:ReservedCodeCacheSize=1024m

# 使用G1垃圾收集器
-XX:+UseG1GC

# 设置G1GC最大暂停时间目标：150毫秒
-XX:MaxGCPauseMillis=150

# G1GC区域大小：16MB（更适合压缩指针）
-XX:G1HeapRegionSize=16m

# G1GC并发回收触发阈值：堆使用率达到40%时启动
-XX:InitiatingHeapOccupancyPercent=40

# 并行垃圾回收线程数：16个线程
-XX:ParallelGCThreads=16

# G1GC并发线程数：8个线程
-XX:ConcGCThreads=8

# JIT编译器线程数量：12个线程
-XX:CICompilerCount=12

# 禁用性能共享内存
-XX:+PerfDisableSharedMem

# 禁用显式GC调用
-XX:+DisableExplicitGC

# 启动时预分配内存
-XX:+AlwaysPreTouch

# 软引用LRU策略：每MB堆内存保留软引用30毫秒
-XX:SoftRefLRUPolicyMSPerMB=30

# 使用压缩普通对象指针：节省内存（最大支持26-28GB）
-XX:+UseCompressedOops

# 使用压缩类指针
-XX:+UseCompressedClassPointers

# 内存溢出时生成堆转储
-XX:+HeapDumpOnOutOfMemoryError

# 堆转储文件路径
-XX:HeapDumpPath=/tmp/android-studio-heapdump.hprof

# 禁止省略堆栈跟踪信息
-XX:-OmitStackTraceInFastThrow

# 忽略无法识别的JVM选项
-XX:+IgnoreUnrecognizedVMOptions

# 文件编码设置
-Dfile.encoding=UTF-8

# 控制台编码设置
-Dconsole.encoding=UTF-8

# 启用子像素抗锯齿
-Dswing.aatext=true

# 字体抗锯齿设置
-Dawt.useSystemAAFontSettings=lcd

# 启用X11渲染后端
-Dsun.java2d.xrender=true

# 设置NIO缓存缓冲区大小
-Djdk.nio.maxCachedBufferSize=4194304

# 使用NIO方式访问ZIP文件
-Djava.util.zip.use.nio.for.zip.file.access=true

# 关闭Kotlin协程调试模式
-Dkotlinx.coroutines.debug=off

# 允许JVM进程附加到自身
-Djdk.attach.allowAttachSelf=true

# 对模块系统非法访问保持静默
-Djdk.module.illegalAccess.silent=true

# 启用断言检查
-ea

# 禁用平台更新检查
-Dide.no.platform.update=true

# 处理取消异常
-Didea.ProcessCanceledException=disabled

# 为AWT使用公平锁
-Dawt.lock.fair=true

# 启用AWT抓取模式
-Dsun.awt.disablegrab=false

# 设置Java为非无头模式
-Djava.awt.headless=false

# 禁用规范路径名缓存
-Dsun.io.useCanonCaches=false

# HTTP隧道认证方案
-Djdk.http.auth.tunneling.disabledSchemes=""

# 限制Attach API只使用临时目录
-Dsun.tools.attach.tmp.only=true

# 启用Kotlin K2编译器
-Didea.kotlin.plugin.use.k2=true

# 启用硬件加速渲染
-Dsun.java2d.accl=true

# 启用并行类加载
-XX:+UseParallelLoadClass

# 启用并行引用处理
-XX:+UseParallelReferences

# 启用字符串去重
-XX:+UseStringDeduplication

使用 gradle 国内镜像

  • Gradle 下载镜像
distributionUrl=https://mirrors.cloud.tencent.com/gradle/gradle-7.2-all.zip
  • 仓库镜像配置
allprojects {
    repositories {
        maven { url 'https://maven.aliyun.com/repository/public/' }
        mavenLocal()
        mavenCentral()
    }
}
